001package edu.pdx.cs410J.cox;
002
003import edu.pdx.cs410J.AbstractPhoneBill;
004
005import java.util.ArrayList;
006import java.util.Collection;
007
008/**
009 * This class represents a phone bill. It is descended from the AbstractPhoneBill class.
010 * It implements AbstractPhoneBill's abstract classes :
011 *      addPhoneCall, getCustomer, and getPhoneCalls.
012 * The following classes are specific to PhoneBill (not inherited):
013 *      setCustomer.
014 * Variables:
015 *      calls: Contains all of the calls made from this account.
016 *      customer: Contains the name of the customer.
017 */
018
019public class PhoneBill extends AbstractPhoneBill<PhoneCall>{
020    private Collection<PhoneCall> calls = new ArrayList<>();
021    private String customer = "notset";
022
023    @Override
024    public void addPhoneCall(PhoneCall call) {
025        /**
026         * This Method accepts a PhoneCall object, which it adds to the existing collection of PhoneCalls.
027         * @param A PhoneCall object which will be added to the collection variable, calls.
028         * */
029        this.calls.add(call);
030    }
031
032    @Override
033    public String getCustomer() {
034        /**
035         * This method returns string containing the name of the customer whose PhoneBill it is.
036         * @return A string which contains the name of the customer who pays the bill.
037         * */
038        return this.customer;
039    }
040
041    @Override
042    public Collection<PhoneCall> getPhoneCalls() {
043        /**
044         * This method returns all of the calls that have taken place for this customer.
045         * @return A collection of PhoneCall instances which hold the information regarding the calls that have taken place.
046         * */
047        return this.calls;
048    }
049
050    public void setCustomer(String name) {
051        /**
052         * This method accepts a string containing the name of the customer whose PhoneBill it is.
053         * @param A string containing the name of the customer who pays the bill.*/
054        this.customer = name;
055    }
056}
