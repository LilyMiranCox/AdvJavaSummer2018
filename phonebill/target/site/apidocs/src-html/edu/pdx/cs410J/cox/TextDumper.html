001package edu.pdx.cs410J.cox;
002
003import edu.pdx.cs410J.PhoneBillDumper;
004import java.io.File;
005import java.io.FileWriter;
006import java.io.IOException;
007import java.io.Writer;
008
009/**
010 * This class deals with writing a completed PhoneBill to a text file
011 */
012
013public class TextDumper implements PhoneBillDumper <PhoneBill>{
014    private String fileName = ""; // A string containing the name of the text file which will be accessed
015
016    /**
017     * This method accepts a PhoneBill, and creates a string with all of its information formatted so that
018     * TextParser can read it in correctly.
019     * @param bill A completed PhoneBill that will have its customer name and PhoneCall information copied into
020     * a string, then written to the specified file
021     */
022    public void dump (PhoneBill bill) {
023        File temp = new File(this.fileName);
024        if(temp.exists() == false) {
025            System.exit(1);
026        }
027        try {
028            Writer billAccess = new FileWriter(temp, false);
029            String toWrite = "!!" + bill.getCustomer() + "!!";
030            for(PhoneCall c: bill.getPhoneCalls()) {
031                toWrite += System.getProperty("line.separator") + c.getCaller() + " " + c.getCallee() + " " + c.getStartTimeString() + " " + c.getEndTimeString();
032            }
033            billAccess.write(toWrite);
034            billAccess.close();
035        } catch (IOException e) {
036            e.printStackTrace();
037        }
038    }
039
040    /**
041     * This method accepts a String containing the name of a text file to be accessed.
042     * @param name A string which contains the name of a text file that will be accessed.
043     */
044    public void setFileName (String name) {
045        this.fileName = name;
046    }
047}
