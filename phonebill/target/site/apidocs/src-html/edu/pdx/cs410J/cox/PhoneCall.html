005/**
006 * This class represents a phone call. It is descended from the AbstractPhoneCall class.
007 * It implements AbstractPhoneCall's abstract base classes:
008 *    getCaller, getCallee, getStartTimeString, and getEndTimeString.
009 * The following methods are specific to PhoneCall:
010 *    setCaller, setCallee, setStartTimeString, setEndTimeString, verifyDateFormat, verifyTimeFormat, verifyPhoneNumberFormat,
011 *    verifyAllDigits.
012 * Variables:
013 *      caller: A string containing the phone number of the person who initiated the call.
014 *      callee: A string containing the phone number of the person who received the call.
015 *      startTime: A string containing the date and time the call was received, separated by a space.
016 *      endTime: A string containing the date and time the call was terminated, separated by a space.
017 */
018
019public class PhoneCall extends AbstractPhoneCall {

021  private String callee = "not implemented";

046     * This method returns the start date/time of the call.

054    /**
055     * This method returns the end date/time of the call.
056     * @return A string containing a date and time separated by a space.

061  public Boolean setCaller(String phoneNum) {
062    /**
063     * This method receives a String containing a phone number, calls verifyPhoneNumberFormat to check

065     * @param A string containing a phone number
066     * @return A boolean value representing whether or not the phone number was formatted correctly, and whether caller was set.

068    boolean phoneNumberCorrect = verifyPhoneNumberFormat(phoneNum);
069    if(phoneNumberCorrect == true) {
070      this.caller = phoneNum;
071    }
072    return phoneNumberCorrect;
073  }

121  public boolean verifyDateFormat(String date) {
122    /**
123     * This method verifies that the received date adheres to the formatting rules. If it doesn't have one or two digits
124     * in the month, separated by '/', one or two digits in the day, separated by '/', and 4 digits in the year, it does
125     * not pass the formatting, and will return false. Otherwise, it is formatted correctly and will return true.
126     * @param A string containing a date
127     * @return True if the date is formatted correctly, false if it is not.
128     * */
129    String trimmedDate = date;
130    int startIndex = 0;
131    int endIndex = 0;
132
133    endIndex = date.indexOf("/");
134    if(endIndex < 0) {
135      System.err.println("Missing at least one '/'.");
136      return false;
137    }
138    trimmedDate = date.substring(startIndex, endIndex);
139
140    // One or two characters (0-9) allowed for month
141    if(verifyAllDigits(trimmedDate) != 1 && verifyAllDigits(trimmedDate) != 2) {
142      System.err.println("Month must contain 1 or 2 digits.");
143      return false;
144    }
145
146    startIndex = endIndex + 1;
147    endIndex = date.indexOf("/", startIndex);
148    if(endIndex < 0) {
149      System.err.println("Missing at least one '/'.");
150      return false;
151    }
152    trimmedDate = date.substring(startIndex, endIndex);
153
154    // One or two characters (0-9) allowed for day
155    if(verifyAllDigits(trimmedDate) != 1 && verifyAllDigits(trimmedDate) != 2) {
156      System.err.println("Day must contain 1 or 2 digits.");
157      return false;
158    }
159
160    startIndex = endIndex + 1;
161    endIndex = date.length();
162    if(endIndex < 0) {
163      System.err.println("Missing a space between the date and time.");
164      return false;
165    }
166    trimmedDate = date.substring(startIndex, endIndex);
167
168    // Four characters (0-9) required for year
169    if(verifyAllDigits(trimmedDate) != 4) {
170      System.err.println("Year must contain 4 digits.");
171      return false;
172    }
173    return true;
174  }

176  public boolean verifyTimeFormat(String time) {
177    /**
178     * This method verifies that the received time adheres to the correct formatting.It must have one or two digits, followed
179     * by ':", followed by two digits. If it not pass the formatting, and will return false. Otherwise, it is formatted correctly and will return true.
180     * @param A string containing a time
181     * @return True if the time is formatted correctly, false if it is not.
182     * */
183    String trimmedDate = time;
184    int startIndex = 0;
185    int endIndex = 0;
186
187    endIndex = time.indexOf(":");
188    if(endIndex < 0) {
189      System.err.println("Missing ':' between hours and minutes.");
190      return false;
191    }
192    trimmedDate = time.substring(startIndex, endIndex);
193
194    // One or two characters (0-9) allowed for hour
195    if(verifyAllDigits(trimmedDate) != 1 && verifyAllDigits(trimmedDate) != 2) {
196      System.err.println("Hour must contain 1 or 2 digits.");
197      return false;
198    }
199    // One semicolon
200    startIndex = endIndex + 1;
201    endIndex = time.length();
202    trimmedDate = time.substring(startIndex, endIndex);
203
204    // Two characters (0-9) required for minutes
205    if(verifyAllDigits(trimmedDate) != 2) {
206      System.err.println("Minutes must contain 1 or 2 digits.");
207      return false;
208    }
209    return true;
210  }

212  public boolean verifyPhoneNumberFormat (String phone) {
213    /**
214     * This method verifies that the received phone number adheres to the correct formatting.It must have three digits, followed
215     * by '-", followed by three digits, followed by '-', followed by 4 digits. If it not pass the formatting, and will return false. Otherwise, it is formatted correctly and will return true.
216     * @param A string containing a phone number
217     * @return True if the phone number is formatted correctly, false if it is not.
218     * */
219    String trimmedPhone = phone;
220    int startIndex = 0;
221    int endIndex = 0;
222
223    endIndex = phone.indexOf("-");
224    if(endIndex < 0) {
225      System.err.println("Missing at least one '-'.");
226      return false;
227    }
228    trimmedPhone = phone.substring(startIndex, endIndex);
229
230    // Three characters (0-9)
231    if(verifyAllDigits(trimmedPhone) != 3) {
232      System.err.println("Phone number is not formatted like ###-###-####.");
233      return false;
234    }
235
236    startIndex = endIndex + 1;
237    endIndex = phone.indexOf("-", startIndex);
238    if(endIndex < 0) {
239      System.err.println("Missing at least one '-'.");
240      return false;
241    }
242    trimmedPhone = phone.substring(startIndex, endIndex);
243
244    // Three characters (0-9)
245    if(verifyAllDigits(trimmedPhone) != 3) {
246      System.err.println("Phone number is not formatted like ###-###-####.");
247      return false;
248    }
249
250    startIndex = endIndex + 1;
251    endIndex = phone.length();
252    if(endIndex < 0) {
253      System.err.println("Missing at least one '-'.");
254      return false;
255    }
256    trimmedPhone = phone.substring(startIndex, endIndex);
257
258    // Four characters (0-9)
259    if(verifyAllDigits(trimmedPhone) != 4) {
260      System.err.println("Phone number is not formatted like ###-###-####.");
261      return false;
262    }
263    return true;
264  }

266  public int verifyAllDigits (String test) {
267    /**
268     * This method tests to make sure that all the characters in a string are digits (0-9)
269     * @param A string that will be tested
270     * @return The number of characters, or 0 if a non digit character is found.
271     */
272    int numDigits = 0;
273    for(int i = 0; i < test.length(); ++i) {
274      if(!Character.isDigit(test.charAt(i))) {
275        return 0;
276      }
277      ++numDigits;
278    }
279    return numDigits;
280  }
