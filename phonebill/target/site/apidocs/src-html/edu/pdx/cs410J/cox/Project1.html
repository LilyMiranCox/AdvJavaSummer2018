013  public static void main(String[] args) {
014    PhoneCall call = new PhoneCall();  // Refer to one of Dave's classes so that we can be sure it is on the classpath
015    PhoneBill bill = new PhoneBill();
016
017    final String README = "\n\n -- README --\n" +
018            "\nProject 1 implemented by Lily Cox \n" +
019            "\nThis program creates phone bills from phone call information." +
020            "\nAll phone call information is entered from the command line in the form: customer callerNumber calleeNumber startTime endTime" +
021            "\nThis information may be preceded by the options: -print or -README" +
022            "\nEach piece of data is then checked to make sure it adheres to the expected formatting." +
023            "\nIf any information is formatted incorrectly, the program will report the error." +
024            "\nIf it passes, however, and the -README option is not used, the information will be added to the instance of the" +
025            "\nPhoneCall class, and that instance added to the specified customer's bill.\n";
026
027    int printOption = 0;
028
029    if(args.length == 0) { // Make sure that arguments have been provided
030      System.err.println("Missing command line arguments");
031      System.exit(1);
032    }
033
034    if(args[0].equals("-README") || args[1].equals("-README")) { // If the -README option is selected
035      System.out.println(README);
036      System.exit(0);
037    }
038
039    if(args[0].equals("-print") || args[1].equals("-print")) { // If the -print option is selected
040      printOption = 1;
041    }
042
043    bill.setCustomer(args[printOption]); // Set the name of the customer in the bill
044    Boolean caller = call.setCaller(args[printOption + 1]); // Test formatting of the phone # and set it
045    Boolean callee = call.setCallee(args[printOption + 2]); // Test formatting of the phone # and set it
046    Boolean start = call.setStartTimeString(args[printOption + 3], args[printOption + 4]); // Test formatting of the date and time and set it
047    Boolean end = call.setEndTimeString(args[printOption + 5], args[printOption + 6]); // Test formatting of the date and time and set it
048    bill.addPhoneCall(call); // Add the added completed phonecall to the phone bill
049
050    if(caller && callee && start && end && printOption == 1) { // If it passes all formatting tests, and all variables are set, print the phone call information
051      System.out.println(call.toString());
052    }
053  }
