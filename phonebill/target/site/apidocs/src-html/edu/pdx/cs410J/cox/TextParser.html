001package edu.pdx.cs410J.cox;
002
003import edu.pdx.cs410J.ParserException;
004import edu.pdx.cs410J.PhoneBillParser;
005import java.io.File;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.util.Scanner;
009
010/**
011 * This class deals with reading and parsing PhoneCall information from text files, and calling the correct methods
012 * to insert the parsed information into PhoneCalls, and each PhoneCall into a PhoneBill.
013 */
014
015public class TextParser implements PhoneBillParser <PhoneBill> {
016    private Boolean formattedCorrectly = false;
017    public String fileName = "";
018
019    /**
020     * This method accepts a String containing the name of an existing text file, checks that it exists, reads and
021     * formats the customer's name, then reads phonecall information and calls setCall to add it to the PhoneBill
022     * @return A PhoneBill filled with all the information (customer name and all PhoneCalls) in the text file,
023     * assuming that the file was formatted correctly.
024     */
025    public PhoneBill parse () {
026
027        if(this.fileName.charAt(this.fileName.length()-4) != '.' || this.fileName.charAt(this.fileName.length()-3) != 't' || this.fileName.charAt(this.fileName.length()-2) != 'x' || this.fileName.charAt(this.fileName.length()-1) != 't') {
028            System.err.println("Included file is not a text file.");
029            System.exit(1);
030        }
031
032        File billAccess = new File(this.fileName);
033        if(billAccess.exists() == false) { // If the attempted file doesn't exist, try to create it
034            try {
035                billAccess.createNewFile();
036            } catch (IOException e) {
037                e.printStackTrace();
038            }
039        }
040        Scanner billFile = null;
041        try {
042            billFile = new Scanner(billAccess); // A Scanner object to read in lines from the file
043        } catch (FileNotFoundException e) {
044            e.printStackTrace();
045        }
046
047        PhoneBill bill = new PhoneBill();
048        if(billFile.hasNextLine() == false) { // If the file is empty
049            System.err.println("Selected file is empty.");
050            System.exit(1);
051        }
052
053        String custName = billFile.nextLine();
054
055        if(custName.length()<5 || custName.charAt(0) != '!' || custName.charAt(1) != '!' || custName.charAt(custName.length()-1) != '!' || custName.charAt(custName.length()-2) != '!') {
056            this.reportBadFormat();
057        }
058        custName = custName.replace("!","");
059        bill.setCustomer(custName);
060
061        while(billFile.hasNextLine() == true) { // For each line of PhoneCall data, call setCall to parse it and add it to the instance of PhoneBill
062            setCall(bill, billFile.nextLine());
063        }
064
065        return bill;
066    }
067
068    /**
069     * This method receives a String containing the name of the file to be accessed
070     * @param name A string containing the name of the file to be accessed, which is used to set the variable fileName.
071     */
072    public void setFileName (String name) {
073        this.fileName = name;
074    }
075
076    /**
077     * This method receives a string containing PhoneCall data that it parses and adds to a PhoneBill if it is formatted correctly.
078     * @param bill An instance of PhoneBill to have a PhoneCall added to it.
079     * @param callInfo A String containing PhoneCall information, that will be parsed and it's information added to a PhoneCall.
080     * @return The received Phonebill instance, with the new PhoneCall data added to it.
081     */
082    public PhoneBill setCall (PhoneBill bill, String callInfo) {
083        PhoneCall call = new PhoneCall();
084        String currentInfo = "";
085
086        // Get the caller number
087        int startIndex = 0;
088        int endIndex = callInfo.indexOf(" ");
089        if(endIndex < 0) {
090            this.reportBadFormat();
091        }
092        currentInfo = callInfo.substring(startIndex, endIndex);
093        Boolean caller = call.setCaller(currentInfo); // Test formatting of the phone # and set it
094
095        // Get the callee number
096        startIndex = endIndex + 1;
097        endIndex = callInfo.indexOf(" ", startIndex);
098        if(endIndex < 0) {
099            this.reportBadFormat();
100        }
101        currentInfo = callInfo.substring(startIndex, endIndex);
102        Boolean callee = call.setCallee(currentInfo); // Test formatting of the phone # and set it
103
104        // Get the start date
105        startIndex = endIndex + 1;
106        endIndex = callInfo.indexOf(" ", startIndex);
107        if(endIndex < 0) {
108            this.reportBadFormat();
109        }
110        currentInfo = callInfo.substring(startIndex, endIndex);
111
112        // Get the start time
113        startIndex = endIndex + 1;
114        endIndex = callInfo.indexOf(" ", startIndex);
115        if(endIndex < 0) {
116            this.reportBadFormat();
117        }
118        String time = callInfo.substring(startIndex, endIndex);
119        Boolean start = call.setStartTimeString(currentInfo, time); // Test formatting of the date and time and set it
120
121        // Get the end date
122        startIndex = endIndex + 1;
123        endIndex = callInfo.indexOf(" ", startIndex);
124        if(endIndex < 0) {
125            this.reportBadFormat();
126        }
127        currentInfo = callInfo.substring(startIndex, endIndex);
128
129        // Get the end time
130        startIndex = endIndex + 1;
131        endIndex = callInfo.length();
132        if(endIndex < 0) {
133            this.reportBadFormat();
134        }
135        time = callInfo.substring(startIndex, endIndex);
136
137        Boolean end = call.setEndTimeString(currentInfo, time); // Test formatting of the date and time and set it
138        bill.addPhoneCall(call); // Add the added completed phonecall to the phone bill
139
140        if(!caller || !callee || !start || !end) { // If the formatting for all are correct
141            this.reportBadFormat();
142        }
143
144        this.formattedCorrectly = true; // Indicate that it has passed all formatting tests
145        return bill;
146    }
147
148    /**
149     * This method returns a Boolean indicating whether or not the text file was formatted correctly.
150     * @return A Boolean indicacting whether or not the text file was formatted correctly.
151     */
152    public Boolean passedFormatting () {
153        return this.formattedCorrectly;
154    }
155
156    /**
157     * This method prints an error indicating that the user named a file that is not formatted correctly, then exits.
158     */
159    public void reportBadFormat () {
160        System.err.println("Included text file is not formatted correctly");
161        System.exit(1);
162    }
163}
