001package edu.pdx.cs410J.cox;

003import edu.pdx.cs410J.AbstractPhoneBill;
004
005import java.io.File;
006import java.util.Arrays;
007
008/**
009 * The main class for the CS410J Phone Bill Project 2
010 */
011
012public class Project2 {
013
014    /**
015     * Facilitates the programs processes. Receives an array of command line arguments, searches for the -print, -README,
016     * and -textFile options, then sends the remaining arguments to check formatting in be set in the PhoneCall class.
017     * If the user uses -textFile and includes a file name, the initial PhoneBill will be loaded with it's contents (assiming
018     * the file is not blank or not formatted incorrectly). At the end of the program, the PhoneBill will be written to the
019     * named file, including any new PhoneCalls that have been added to it.
020     */
021    public static void main(String[] args) {
022        PhoneCall call = new PhoneCall();  // Refer to one of Dave's classes so that we can be sure it is on the classpath
023        PhoneBill bill = new PhoneBill();
024
025        final String README = "\n\n -- README --\n" +
026                "\nProject 2 implemented by Lily Cox \n" +
027                "\nThis program creates phone bills from phone call information." +
028                "\nAll phone call information is entered in the form: customer callerNumber calleeNumber startTime endTime" +
029                "\nInformation can be brought in from a file included in the command line arguments and from the command line itself" +
030                "\nCall information in the command line may be preceded by the options: -print, -README, or -textFile followed by the name of the file." +
031                "\nEach piece of data is then checked to make sure it adheres to the expected formatting." +
032                "\nIf any information is formatted incorrectly, the program will report the error." +
033                "\nIf it passes, however, and the -README option is not used, the information will be added to the instance of the" +
034                "\nPhoneCall class, and that instance added to the specified customer's bill." +
035                "\nIf a file is used but the names in the file and command line do not match, an error will be reported\n";
036
037        int numOptionArgs = 4; // The max number of argument positions that the available options could take up
038        int numOptionsUsed = 0; // The number of argument positions actually used by user options
039        Boolean useFile = false; // If the user includes the '-textFile' option with a valid file name
040        String billFileName = ""; // File access for the user included file name
041        Boolean printCall = false; // If the user includes the '-print' option
042        TextParser readFile = new TextParser();
043
044        if(args.length == 0) { // Checks that command line arguments have been provided
045            System.err.println("Missing command line arguments");
046            System.exit(1);
047        }
048
049        for(int i = 0; i < numOptionArgs; ++i) { // Loop through to find how many options are used so we know how many positions to offset the phone call information
050            if(args[i].equals("-README")) { // If user wants readme, even if there are other options, only print readme and exit
051                printReadme(README);
052            }
053            else if(args[i].equals("-print")) { // If user wants to print the phonecalls in the bill, make the Boolean indicating this choice, true
054                if(printCall == true) {
055                    System.err.println("-print option used multiple times.");
056                    System.exit(1);
057                }
058                printCall = true;
059                ++numOptionsUsed;
060            }
061            else if(args[i].equals("-textFile")) { // If user wants to read/write to an existing text file, check that the file exists, make the Boolean indicating this choice, true
062                if(useFile == true) {
063                    System.err.println("-textFile option used multiple times.");
064                    System.exit(1);
065                }
066                File billAccess = new File(args[i+1]); // Try to access the requested file
067                billFileName = args[i+1];
068                useFile = true;
069                numOptionsUsed +=2; // It counts for two array positions since one is '-textFile' and the other is the name of the file
070            }
071        }
072
073        if(args.length > (numOptionsUsed + 7)) { // If too many phone call arguments were included
074            System.err.println("Too many phone call arguments.");
075            System.exit(1);
076        }
077
078        if(args.length != (numOptionsUsed + 7)) { // Since there are seven total pieces of data expected for the PhoneCall, there must be that many arguments (if no options are used)
079            System.err.println("Does not contain all required phone call arguments.");
080            System.exit(1);
081        }
082
083        if(useFile == true) { // If the user wants to read/write from a file, read the information, parse it, and insert it into an instance of PhoneBill, which is returned
084            readFile.setFileName(billFileName);
085            bill = readFile.parse();
086        }
087
088        if(bill.getCustomer().equals(args[numOptionsUsed]) == false && useFile == true && bill.getCustomer().equals("notset") == false ) { // If the name in the file doesn't match the name of the name of the PhoneCall to add from the command line
089            System.err.println("Received customer name does not match customer name in the text file.");
090            System.exit(1);
091        }
092
093        // Set the PhoneCall information, and add it to the PhoneBill
094        bill.setCustomer(args[numOptionsUsed]); // Set the name of the customer in the bill
095        Boolean caller = call.setCaller(args[numOptionsUsed + 1]); // Test formatting of the phone # and set it
096        Boolean callee = call.setCallee(args[numOptionsUsed + 2]); // Test formatting of the phone # and set it
097        Boolean start = call.setStartTimeString(args[numOptionsUsed + 3], args[numOptionsUsed + 4]); // Test formatting of the date and time and set it
098        Boolean end = call.setEndTimeString(args[numOptionsUsed + 5], args[numOptionsUsed + 6]); // Test formatting of the date and time and set it
099        bill.addPhoneCall(call); // Add the added completed phonecall to the phone bill
100
101        if((caller && callee && start && end)) { // If it passes all formatting tests, and all variables are set
102            if((useFile == true && readFile.passedFormatting() == true) || useFile == false) {
103                if (printCall == true) {
104                    System.out.println("Bill for: " + bill.getCustomer());
105                    for (PhoneCall c : bill.getPhoneCalls()) {
106                        printPhonecall(c);
107                    }
108                }
109
110                if (useFile == true) { // If the user wanted to read/write from a file, write all information in the PhoneBill to the file, including added PhoneCall
111                    TextDumper saveFile = new TextDumper();
112                    saveFile.setFileName(billFileName);
113                    saveFile.dump(bill);
114                }
115            }
116        }
117    }
118
119    /**
120     * This method accepts a String containing a README statement, prints the statement to the console, then exits.
121     * @param README A String containing a README statement
122     */
123    public static void printReadme (String README) {
124        System.out.println(README);
125        System.exit(0);
126    }
127
128    /**
129     * This method receives a PhoneCall, and prints out its contents to the console
130     * @param call An instance of PhoneCall that will have it's information printed to the console
131     */
132    public static void printPhonecall (PhoneCall call) {
133
134        System.out.println(call.toString());
135    }
136}
